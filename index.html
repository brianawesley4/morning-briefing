// api/pipeline.js // Reads from the Estem CRM (estem-crm.vercel.app) — the source of truth. // Source of truth: Estem CRM (estem-crm.vercel.app) export default async function handler(req, res) { res.setHeader('Content-Type', 'application/json'); res.setHeader('Cache-Control', 'no-store'); res.setHeader('Access-Control-Allow-Origin', '*'); res.setHeader('Access-Control-Allow-Methods', 'GET, OPTIONS'); if (req.method === 'OPTIONS') return res.status(200).end(); // ── 1. Fetch all leads from the Estem CRM ────────────────────── const crmRes = await fetch('https://estem-crm.vercel.app/api/leads', { headers: { 'Accept': 'application/json' } }); if (!crmRes.ok) { return res.status(502).json({ ok: false, error: `CRM returned HTTP ${crmRes.status}` }); } const leads = await crmRes.json(); if (!Array.isArray(leads)) { return res.status(502).json({ ok: false, error: 'CRM did not return a lead array' }); } const active = leads.filter(l => !l.is_archived); // ── 2. Pipeline counts by stage ──────────────────────────────── const stageCounts = {}; active.forEach(l => { const s = l.stage || 'Unknown'; stageCounts[s] = (stageCounts[s] || 0) + 1; }); // ── 3. Hot leads — Grade A/A+ with notes ───────────────────── const hotLeads = active .filter(l => ['A+', 'A'].includes(l.grade) && l.notes?.trim()) .sort((a, b) => (b.score || 0) - (a.score || 0)) .slice(0, 8) .map(l => ({ id: l.id, name: l.name, stage: l.stage, grade: l.grade, score: l.score, phone: l.phone, email: l.email, notes: l.notes, daysSinceContact: l.lcd, source: l.source, type: l.type, assigned: l.assigned })); // ── 4. Overdue — A/A+/B grade, not contacted in 90+ days ───── const overdueLeads = active .filter(l => ['A+', 'A', 'B'].includes(l.grade) && (l.lcd || 0) >= 90) .sort((a, b) => (b.score || 0) - (a.score || 0)) .slice(0, 10) .map(l => ({ name: l.name, stage: l.stage, grade: l.grade, daysSinceContact: l.lcd, phone: l.phone, notes: l.notes?.slice(0, 100) })); // ── 5. Summary pipeline counts ──────────────────────────────── const buyers = active.filter(l => l.type?.includes('Buyer')).length; const sellers = active.filter(l => l.type?.includes('Seller')).length; const sphere = active.filter(l => l.stage === 'Sphere').length; const referrals = active.filter(l => l.stage === 'Referral').length; const pastClients = active.filter(l => l.stage === 'Past Client').length; const highGrade = active.filter(l => ['A+', 'A'].includes(l.grade)).length; const hot = active.filter(l => l.stage === 'Hot').length; const revived = active.filter(l => l.is_revived === true).length; // FIXED: overdueCount = A/A+/B grade leads with lcd >= 90 — was incorrectly using highGrade const overdueCount = active.filter(l => ['A+','A','B'].includes(l.grade) && (l.lcd||0) >= 90).length; const closedThisYear = active.filter(l => l.stage === 'Closed').length; const newToday = active.filter(l => { if (!l.created_at) return false; const d = new Date(l.created_at); const now = new Date(); return d.toDateString() === now.toDateString(); }).length; // ── 6. Source breakdown for AI context ──────────────────────── const sourceCounts = {}; active.forEach(l => { const s = l.source || 'Unknown'; sourceCounts[s] = (sourceCounts[s] || 0) + 1; }); const crmContext = [ `Total active leads: ${active.length}`, `Grade A+/A leads: ${highGrade} (${overdueLeads.length} not contacted in 90+ days)`, `Buyers: ${buyers} | Sellers: ${sellers} | Sphere: ${sphere} | Referrals: ${referrals}`, `Past Clients: ${pastClients} | Closed: ${closedThisYear}`, `Top hot lead: ${hotLeads[0]?.name || 'none'} (${hotLeads[0]?.daysSinceContact || 0}d since contact)`, `Sources: ${Object.entries(sourceCounts).map(([k,v])=>`${k}:${v}`).join(', ')}` ].join(' | '); return res.status(200).json({ ok: true, lastUpdated: new Date().toISOString(), // Pipeline counts for Revenue Command card pipeline: { totalActive: active.length, buyers, sellers, sphere, referrals, pastClients, highGrade, hot, revived, overdue: overdueCount, closedThisYear, newToday, stageCounts }, // Hot leads for Follow-Up card hotLeadsList: hotLeads, // Overdue contacts overdueList: overdueLeads, // For AI summary crmContext, sources: { crm: true, } }); } catch (err) { return res.status(500).json({ ok: false, error: err.message }); } }
